example.py
from incagent import IncAgent, Contract, ContractTerms from incagent import NegotiationPolicy from incagent.messaging import MessageBus bus = MessageBus() vendor = IncAgent(name="Vendor", role="seller", message_bus=bus) buyer = IncAgent(name="Buyer", role="buyer", message_bus=bus) contract = Contract( title="Outbound Campaign", terms=ContractTerms(quantity=1, unit_price_range=(4500, 5500)), ) result = await buyer.negotiate(contract, counterparty=vendor) print(result.status) # agreed print(result.final_terms) # $5,000
